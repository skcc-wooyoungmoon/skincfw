에제 1. lovPopup (단일건 선택) 
    SKIAF.util.lovPopup({
        codeGroupId : $('#codeGroupId').val(),
        keyword : $('#codeDetailId1').val()
    }, function(data) {
        $('#codeDetailDataArea1').setData(data[0]);
    });
에제 2. lovPopup (복수건 선택) 
                            
    SKIAF.util.lovPopup({
        codeGroupId : $('#codeGroupId').val(),
        singleSelect : false
    }, function(data) {
        $('#codeDetailDataArea3').setData({
            data : JSON.stringify(data)
        });
    });
